## Ed25519 Signature Verification Steps

### Verification Process Overview

1. Verify that the SHA-256 hash of `solution_*.bin` matches the `sha256_hash` in the `poc_result_*.json`.
2. Use `public_key.pem` to verify the Ed25519 signature stored in the `poc_result_*.json` (`signature` field).
3. If both checks pass → The result was generated by Quantum Alpha Technology Co., Ltd. and has not been tampered with.

## Verification Command Examples

### Python Example (Recommended)

```python
from cryptography.hazmat.primitives.asymmetric import ed25519
from cryptography.hazmat.primitives import serialization
import hashlib
import json
import base64
from pathlib import Path

def verify_poc_result(json_path: str, bin_path: str, public_key_pem: str):
    with open(json_path, 'r', encoding='utf-8') as f:
        result = json.load(f)

    exec_sum = result.get("executive_summary", {})
    signature_b64 = result.get("signature")
    sha256_expected = result.get("sha256_hash")

    if not signature_b64:
        print("❌ Missing signature field in JSON")
        return False

    # 1. Verify SHA-256
    data = Path(bin_path).read_bytes()
    sha256_actual = hashlib.sha256(data).hexdigest()

    if sha256_actual != sha256_expected:
        print(f"❌ SHA-256 mismatch! Expected: {sha256_expected}, Actual: {sha256_actual}")
        return False
    print("✅ SHA-256 hash verified successfully")

    # 2. Ed25519 signature verification
    pub = serialization.load_pem_public_key(Path(public_key_pem).read_bytes())
    if not isinstance(pub, ed25519.Ed25519PublicKey):
        print("❌ public_key.pem is not an Ed25519 public key")
        return False

    try:
        pub.verify(base64.b64decode(signature_b64), data)
        print("✅ Ed25519 signature verification passed")
        print(f"   Task: {exec_sum.get('task_type')}")
        print(f"   Status: {exec_sum.get('status')}")
        print(f"   true_conflicts: {exec_sum.get('true_conflicts')}")
        return True
    except Exception as e:
        print(f"❌ Ed25519 verification failed: {e}")
        return False

# Usage example
if __name__ == "__main__":
    verify_poc_result(
        json_path="poc_result_steiner_tree_63_202607xx.json",
        bin_path="solution_steiner_tree_63bit_seed42.bin",
        public_key_pem="keys/public_key.pem"
    )
```

### Bash Quick Verification

```bash
#!/bin/bash
JSON_FILE=$1
BIN_FILE=$2
PUBKEY="keys/public_key.pem"

ACTUAL_HASH=$(sha256sum "$BIN_FILE" | awk '{print $1}')
EXPECTED_HASH=$(jq -r '.sha256_hash' "$JSON_FILE")

if [ "$ACTUAL_HASH" != "$EXPECTED_HASH" ]; then
    echo "❌ SHA-256 mismatch"
    exit 1
fi
echo "✅ SHA-256 verified"

python3 -c "
import json, base64
from cryptography.hazmat.primitives.asymmetric import ed25519
from cryptography.hazmat.primitives import serialization
from pathlib import Path

data = Path('$BIN_FILE').read_bytes()
result = json.load(open('$JSON_FILE'))
sig = base64.b64decode(result['signature'])
pub = serialization.load_pem_public_key(Path('$PUBKEY').read_bytes())
pub.verify(sig, data)
print('✅ Ed25519 signature verified successfully')
"
```
